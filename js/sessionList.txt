app.session={
    array :(k,v)=>{
        sessionStorage.setItem(k, JSON.stringify(v));
    },
    getArray : k=>{
        return JSON.parse(sessionStorage.getItem(k));
    },
    init : (k,v) => {
        sessionStorage.setItem(k,v);
    },
    getSessionData : k => {
        return sessionStorage.getItem(k);
    }
};
